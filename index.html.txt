function renderFilms(){
  renderFeatured();
  const q=(document.getElementById('fsearch')||{value:''}).value.toLowerCase();
  const sort=(document.getElementById('fsort')||{value:'recent'}).value;
  let list=films.filter(f=>(filmFilter==='tous'||f.type===filmFilter)&&(!q||f.title.toLowerCase().includes(q)||f.desc.toLowerCase().includes(q)));
  if(sort==='alpha')list.sort((a,b)=>a.title.localeCompare(b.title));
  else if(sort==='rating')list.sort((a,b)=>b.rating-a.rating);
  else list.sort((a,b)=>b.id-a.id);
  const g=document.getElementById('fgrid');
  const c=document.getElementById('fcnt');
  const s=document.getElementById('fstat-tot');
  if(s)s.textContent=films.length;
  if(c)c.textContent=list.length+' résultat'+(list.length>1?'s':'');
  if(!g)return;
  if(!list.length){g.innerHTML=`<div class="empty"><div class="empty-icon">🎬</div><p>Aucun contenu.${isAdmin?' Ajoutez un film !':''}</p></div>`;return;}
  g.innerHTML=list.map(f=>`
    <div class="fcard" onclick="ofd(${f.id})">
      <div class="fthumb">
        <span class="femoji">${f.emoji}</span>
        <div class="fovr"><div class="fplay">▶</div></div>
        <span class="fbadge fb-free">Gratuit</span>
        ${f.isNew?'<span class="fbadge fb-new">Nouveau</span>':''}
        ${f.type==='serie'?'<span class="fbadge fb-serie">Série</span>':''}
      </div>
      <div class="finfo">
        <div class="fname">${f.title}</div>
        <div class="fmeta"><span>${f.year}</span><span>•</span><span>${f.duration}</span><span style="color:#fbbf24">${'★'.repeat(f.rating)}</span></div>
        <span class="fcat">${f.type}</span>

      </div>
    </div>`).join('');
}

function openAF(id=null){
  editFId=id;
  document.getElementById('afilm-title').textContent=id?'Modifier le film':'Ajouter un film';
  if(id){
    const f=films.find(x=>x.id===id);
    document.getElementById('af-ti').value=f.title;
    document.getElementById('af-ty').value=f.type;
    document.getElementById('af-yr').value=f.year;
    document.getElementById('af-de').value=f.desc;
    document.getElementById('af-du').value=f.duration;
    document.getElementById('af-ra').value=f.rating;
    document.getElementById('af-em').value=f.emoji;
    document.getElementById('af-li').value=f.link;
    document.getElementById('af-l2').value=f.link2||'';
    document.getElementById('af-ne').value=f.isNew?'true':'false';
    document.getElementById('af-fe').value=f.featured?'true':'false';
  } else {
    ['af-ti','af-yr','af-de','af-du','af-li','af-l2'].forEach(i=>document.getElementById(i).value='');
    document.getElementById('af-ty').value='film';document.getElementById('af-ra').value='5';
    document.getElementById('af-em').value='✝️';document.getElementById('af-ne').value='true';document.getElementById('af-fe').value='false';
  }
  document.getElementById('afilm-modal').classList.add('open');
}
function saveFilm(){
  const ti=document.getElementById('af-ti').value.trim();
  if(!ti){toast('⚠ Entrez un titre','g');return;}
  const d={title:ti,type:document.getElementById('af-ty').value,year:+document.getElementById('af-yr').value||2024,desc:document.getElementById('af-de').value,duration:document.getElementById('af-du').value||'—',rating:+document.getElementById('af-ra').value,emoji:document.getElementById('af-em').value,link:document.getElementById('af-li').value,link2:document.getElementById('af-l2').value,isNew:document.getElementById('af-ne').value==='true',featured:document.getElementById('af-fe').value==='true'};
  if(d.featured)films.forEach(f=>f.featured=false);
  if(editFId){const i=films.findIndex(f=>f.id===editFId);films[i]={...films[i],...d};toast('✓ Film modifié !','v');}
  else{d.id=Date.now();films.unshift(d);toast('✓ Film ajouté !','v');}
  saveF();document.getElementById('afilm-modal').classList.remove('open');renderFilms();
}
function delFilm(id){if(!confirm('Supprimer ce film ?'))return;films=films.filter(f=>f.id!==id);saveF();renderFilms();toast('🗑 Film supprimé','v');}
